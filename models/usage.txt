IMPORTS

from models import Course, Hole, Tee, HoleScore, Round, User


CREATING A COURSE
-----------------
hole = Hole(number=1, par=4, handicap=7)

tee = Tee(
    color="blue",
    slope_rating=128,

    hole_yardages={1: 425, 2: 380, ...}
)

course = Course(
    id="pebble-beach",
    name="Pebble Beach",
    location="Pebble Beach, CA",
    par=72,  # or leave None to calculate from holes
    holes=[hole1, hole2, ...],  # 18 Hole objects in order
    tees=[tee1, tee2, ...]
)

# Access
course.get_hole(5)           # returns Hole at position 5
course.get_tee("blue")       # returns Tee by color
course.get_par()             # returns direct par or calculates from holes
course.front_nine_par        # calculated property
course.back_nine_par         # calculated property

UPDATING FIELDS (for scanner corrections)
-----------------------------------------
# All models inherit update_field() which validates on update
error = course.update_field("par", 72)
if error:
    print(f"Invalid: {error}")

# Direct assignment also validates (raises ValidationError)
course.par = 72


KEY POINTS
----------
- All fields are Optional for handling incomplete scanned data
- Lists (holes, tees, hole_scores, rounds) are stored IN ORDER by hole number
- Use get_xxx() methods for "direct value or calculate" pattern
- Round connects HoleScores to Course for par-based calculations
- HoleScore.get_score_type(par) needs par passed in from Round
